import cadquery as cq
import math
from cadquery import exporters

# --- Part 1: Rectangular Plate ---
sketch_scale = 0.75
length = 0.75 * sketch_scale
width = 0.375 * sketch_scale
height = 0.0063

# Define corner radius (scaled)
corner_radius = min(length, width) / 10  # Adjust as needed

# Create the base rectangle with rounded corners
part_1 = (
    cq.Workplane("XY")
    .moveTo(0, corner_radius)
    .threePointArc((corner_radius, 0), (0, corner_radius))
    .lineTo(length - corner_radius, 0)
    .threePointArc((length, corner_radius), (length - 0.5 * corner_radius, width))
    .lineTo(length, width - corner_radius)
    .threePointArc((length - corner_radius, width), (length - 0.5 * corner_radius, width - 0.375 * corner_radius))
    .lineTo(corner_radius, width)
    .threePointArc((0, width - corner_radius), (0, width - corner_radius))
    .close()
    .extrude(height)
)

# Export to STL
exporters.export(part_1, './stlcq/0043/00435339.stl')

# --- Assembly (if needed, based on the single part) ---
# In this case, there's only one part, so no assembly is needed.
# If there were multiple parts, you would use .union() or .cut() to combine them.

# --- Final Result ---
result = part_1
