The final assembly combines these two parts to create the desired structure.
import cadquery as cq

# --- Part 1: Vertical Rod ---
part_1 = (
    cq.Workplane("XY")
    .moveTo(0.0, 0.0)
    .lineTo(0.0187 * 0.75, 0.0)
    .lineTo(0.0187 * 0.75, 0.0187 * 0.75)
    .lineTo(0.7354 * 0.75, 0.0187 * 0.75)
    .lineTo(0.7354 * 0.75, 0.6136 * 0.75)
    .lineTo(0.0187 * 0.75, 0.6136 * 0.75)
    .lineTo(0.0187 * 0.75, 0.75 * 0.75)
    .lineTo(0.0, 0.75 * 0.75)
    .lineTo(0.0, 0.0)
    .close()
    .extrude(-0.0188)
)

# --- Coordinate System Transformation for Part 1 ---
part_1 = part_1.rotate((0, 0, 0), (0, 0, 1), -90)
part_1 = part_1.translate((0, 0, 0))

# --- Part 2: Cube ---
part_2 = (
    cq.Workplane("XY")
    .rect(0.7354 * 0.7354, 0.75 * 0.7354)
    .extrude(0.0188)
)

# --- Coordinate System Transformation for Part 2 ---
part_2 = part_2.rotate((0, 0, 0), (0, 0, 1), -90)
part_2 = part_2.translate((0.0187, 0, 0.0187))

# --- Assembly ---
assembly = part_1.union(part_2)

cq.exporters.export(assembly, "./stlcq/0035/00352431.stl")
cq.exporters.export(assembly, "./stlcq/00359903.stl
